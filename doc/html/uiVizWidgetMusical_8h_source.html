540 void loadChords(TiXmlElement* musicDataElm) {
541 if (musicDataElm == NULL) return;
542 TiXmlElement* selectedChordsElm = musicDataElm->FirstChildElement("selectedChords");
543 if (selectedChordsElm != NULL) {
544
545 clearSelectedChords();
546
547 // Chords ...
548 int key = 1;
549 for(TiXmlNode* child = selectedChordsElm->FirstChild(); child; child = child->NextSibling() ) {
550 TiXmlPrinter printer;
551 child->Accept( &printer );
552 string elmXML = printer.CStr();
553 vizChord chord = vizChord(elmXML, false);
554 addSelectedChord(chord, false, false);
555
556 // Set selected key to 1st chord if no Key set...
557 if (key == 1 && getSelectedKey().getNoteName() == "") setSelectedKey(chord.getChordRootNote());
558
559 // We only want to apply rules if getIsUserDefinedPoints is false
560 if (!chord.getIsUserDefinedPoints()) {
561 uiVizInstrumentRuleArgs ruleArgs = uiVizInstrumentRuleArgs(chord.getInstrumentRules());
562 onRulesChanged(ruleArgs);
563 }
564
565 key++;
566 }
