677 virtual void updateFrets(float sizingRatio)
678 {
679 mFrets.clear();
680
681 float fretWidth = scale(mFRET_SIZE) * sizingRatio;
682 float sizeToUse = getIsFixedFretWidth() ? scale(getUsableFretboardLength()) : scale(getSize());
683
684 float neckRemainder = 0.0f;
685 if (mUseFullNeck)
686 {
687 neckRemainder = getNeckRemainder();
688 sizeToUse = getIsFixedFretWidth() ? sizeToUse : sizeToUse + neckRemainder;
689 }
690
691 float oldFretDist = sizeToUse;
692
693 /***********************************************************************************************
694 The number of frets past the neckRemainder (in lefty mode!). At the given ratio, 4 frets get
695 chopped off, so we subtract the neckRemainder and 4 * fretWidth from the fret x position!
696 JC helped me with this :)
697 ***********************************************************************************************/
698 float magicFretMultiplier = 4.0f;
699
700 int lowestStartingFret = getLowestStartingFret();
701 int highestEndingFret = getHighestEndingFret();
702
703 for (int i = 1; i <= mTotalFrets; i++)
704 {
705 float newFretDist = (getIsFixedFretWidth() ? oldFretDist - (sizeToUse / (float)mTotalFrets) : oldFretDist / FRET_RATIO);
706
707 float fretPos;
708
709 switch (getTheoryVizOrientationMode())
710 {
711 case TheoryVizOrientationMode::HORIZONTAL:
712 fretPos = isLefty() ? (getIsFixedFretWidth() ? newFretDist - fretWidth * 2 : newFretDist - (magicFretMultiplier * fretWidth) - neckRemainder) : sizeToUse + (newFretDist * -1);
713 break;
714 case TheoryVizOrientationMode::VERTICAL:
715 fretPos = sizeToUse + (newFretDist * -1); // <--- this works!!
716 break;
717 }
718
719 uiVizWigetStringedInstrumentFret f = uiVizWigetStringedInstrumentFret();
720 f.prevFretDist = oldFretDist;
721 f.fretDist = newFretDist;
722 f.fretPos = isLefty() ? fretPos : fretPos + fretWidth;
723
724 bool isInFretRange = getIsChordDiagramMode() ? (i + lowestStartingFret - 1 <= highestEndingFret) : (i >= lowestStartingFret - 1 && i <= highestEndingFret);
725
726 ofColor fretColor = getFretColor(isInFretRange);
727
728 switch (getTheoryVizOrientationMode())
729 {
730 case TheoryVizOrientationMode::HORIZONTAL:
731 f.path.setRectangle(scale(getUsableX()) + fretPos + fretWidth, scale(getUsableY()), fretWidth, scale(getUsableHeight()), fretColor);
732
733 f.fretArea.setRectangle(
734 isLefty() ? scale(getUsableX()) + fretPos + fretWidth * 2 : scale(getUsableX()) + fretPos - (oldFretDist - newFretDist) + fretWidth * 2,
735 scale(getUsableY()),
736 isLefty() ? (oldFretDist - newFretDist) - fretWidth : oldFretDist - newFretDist - fretWidth, // scale(getUsableX()) + fretPos - fretWidth,
737 scale(getUsableHeight()), getFretColor(true));
738 break;
739 case TheoryVizOrientationMode::VERTICAL:
