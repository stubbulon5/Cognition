82

85 
86  ofColor color = vizTheory::getColorForKey(slice->note.getNoteName());
87 
88  //---------- Draw a slice ----------
89  slice->regularColor = ofColor::fromHsb(color.getHue(), 220, 220);

96 
97  ofPath c;

99  /* Expensive operation: Keep OUT of draw cycle */
100 
101  c.moveTo(circX, circY);
102  c.arc(circX, circY, circRadiusX, circRadiusY, (float)currentAngle + (float)angleOffset, degreesPerSlice + (float)angleOffset);
103  c.close();
104  slice->regularColor = ofColor::fromHsb(color.getHue(), 220, 220);
105  slice->path.update(c);
106 
107  //---------- Label plot calculations for slice ----------
108  float midSlice = ((float)degreesPerSlice/2) + (float)currentAngle + (float)angleOffset;
109  float distanceFromCentre = 0.75;
110  ofRectangle rect = font->rect(slice->note.getNoteName());
111  float fontWidth = rect.width;
112  float fontHeight = rect.height;
113 
114  // Now that we have the middle of the slice angle, we use cos, sin on it, and multiply a distanceFromCentre factor on the angle to work
115  // out a label coordinate. Finally we have the centre of the font to the x, y coords!
116 
117  float plotX = (float)(circX - fontWidth /(float) 2) + circRadiusX * distanceFromCentre * (float)cos((float)ofDegToRad(midSlice));
118  float plotY = (float)(circY + fontHeight/(float) 2) + circRadiusY * distanceFromCentre * (float)sin((float)ofDegToRad(midSlice));
119  slice->labelPoint = ofPoint(plotX, plotY);
120  }
121

123  circleInner.setFilled(true);
124  circleInner.arc(circX, circY, circRadiusX/(float)2, circRadiusY/(float)2, 0, 360);
