65 void update(uiVizWidgetContext context) override {
66 font = getFontSizedForDimensions(getUsableWidth(), getUsableHeight());
67 fontLargeLabel = getLargeFontSizedForDimensions(getUsableWidth(), getUsableHeight());
68 rectFontHitArea = font->rect("G#");
69 circleInner.clear();
70
71 float circX = scale(getUsableX() + getUsableWidth()/2.0f);
72 float circY = scale(getUsableY() + getUsableHeight()/2.0f);
73 float circRadiusX = scale(getUsableWidth() /2.0f);
74 float circRadiusY = scale(getUsableHeight()/2.0f);
75
76 float degreesPerSlice = (float)360 / (float)keys.size();
77 float currentAngle = 0;
78
79 for (int i = 1; i <= circleSlices.size() ; i++) {
80
81 uiVizWidgetKeyPickerSlice* slice = &circleSlices[i-1];
82
83 // The current degrees position
84 currentAngle = ((float)degreesPerSlice*i);
85
86 ofColor color = vizTheory::getColorForKey(slice->note.getNoteName());
87
88
89 //---------- Draw a slice ----------
90 slice->regularColor = ofColor::fromHsb(color.getHue(), getSelectedKey().getNoteName() == "" ? 200 : 100, 200);
91 slice->hoveredColor = ofColor::fromHsb(color.getHue(), 220, 255);
92 slice->selectedColor = ofColor::fromHsb(color.getHue(), 220, 255);
93
94 slice->regularFontColor = getSelectedFontColor(slice->note.getNoteName());
95 slice->hoveredFontColor = getSelectedFontColor(slice->note.getNoteName());
96 slice->selectedFontColor = getSelectedFontColor(slice->note.getNoteName());
97
98 ofPath c;
99 c.setCircleResolution(360);
100 /* Expensive operation: Keep OUT of draw cycle */
101
102 c.moveTo(circX, circY);
103 c.arc(circX, circY, circRadiusX, circRadiusY, (float)currentAngle + (float)angleOffset, degreesPerSlice + (float)angleOffset);
104 c.close();
105 // slice->regularColor = ofColor::fromHsb(color.getHue(), 220, 220);
106 slice->path.update(c);
107
108 //---------- Label plot calculations for slice ----------
109 float midSlice = ((float)degreesPerSlice/2) + (float)currentAngle + (float)angleOffset;
110 float distanceFromCentre = 0.75;
111 ofRectangle rect = font->rect(slice->note.getNoteName());
112 float fontWidth = rect.width;
113 float fontHeight = rect.height;
114
115 // Now that we have the middle of the slice angle, we use cos, sin on it, and multiply a distanceFromCentre factor on the angle to work
116 // out a label coordinate. Finally we have the centre of the font to the x, y coords!
117
118 float plotX = (float)(circX - fontWidth /(float) 2) + circRadiusX * distanceFromCentre * (float)cos((float)ofDegToRad(midSlice));
119 float plotY = (float)(circY + fontHeight/(float) 2) + circRadiusY * distanceFromCentre * (float)sin((float)ofDegToRad(midSlice));
120 slice->labelPoint = ofPoint(plotX, plotY);
121 }
122
123 circleInner.setCircleResolution(360); // Can lower or remove to bump up performance
124 circleInner.setFilled(true);
125 circleInner.arc(circX, circY, circRadiusX/(float)2, circRadiusY/(float)2, 0, 360);
126 vizElm_circleInner.update(circleInner);
127 circleInnerPoint.set(circX, circY);
128 }

130 void draw(uiVizWidgetContext context) override {
131 // Draw the slices
132 bool anyNoteHovered = false;
133 for(uiVizWidgetKeyPickerSlice& slice:circleSlices){
134 // uiVizWidgetKeyPickerSlice& slice = circleSlices[i-1];
135
136 slice.path.setColor(slice.regularColor);
137
138 //---------- Mouse Hit test ----------
139 float dist = ofDist(ofGetMouseX(), ofGetMouseY(), slice.labelPoint.x, slice.labelPoint.y);
140 bool sliceHovered = dist <= ((rectFontHitArea.width*1.5f + rectFontHitArea.height*1.5f) / (float)2);
141 slice.isHovered = false;
142 slice.isSelected = false;
143 bool doesMatchSelectedKey = matchesSelectedKey(slice.note);
144 bool didSelectKey = false;
145
146 if (sliceHovered || doesMatchSelectedKey) {
147
148 didSelectKey = (sliceHovered && getIsMousePressedOverWidget(false));
149 if (doesMatchSelectedKey || didSelectKey) {
150 slice.isSelected = true;
151 slice.path.setColor(slice.selectedColor);
152 vizElm_circleInner.setColor(slice.selectedColor);
153
154 if (didSelectKey) {
155 setSelectedKey(slice.note);
156 uiVizNoteSelectedArgs args("Key Picker", slice.note);
157 ofNotifyEvent(keySelected, args);
158 }
159 } else {
160 slice.path.setColor(slice.hoveredColor);
161 if (getSelectedKey().getNoteName() == "") vizElm_circleInner.setColor(slice.hoveredColor);
162 slice.isHovered = true;
163 setHoveredKey(slice.note);
164 if (sliceHovered) {
165 anyNoteHovered = true;
166 auditionNote(slice.note);
167 }
168 }
169 }
170
171 slice.path.draw();
172 // Event Stuff
173
174
175 ofPushStyle();
176 ofSetColor(getLabelStateColor(slice));
177 font->draw(slice.note.getNoteName(), slice.labelPoint.x, slice.labelPoint.y);
178 ofPopStyle();
179 //if (!anyNoteHovered) clearHoveredKey();
180 }
181
182
183 // Draw the inner dot
184 if (getSelectedKey().getNoteName() != "") {
185 vizElm_circleInner.draw();
186
187 ofPushStyle();
188 ofSetColor(vizTheory::getColorForKey(getSelectedKey().getNoteName()), 255);
189 rectLargeLabel = fontLargeLabel->rect(getSelectedKey().getNoteName());
190 drawFontWithShadow(
191 fontLargeLabel, getSelectedKey().getNoteName(),
192 circleInnerPoint.x - rectLargeLabel.width / 2.0f,
193 circleInnerPoint.y + rectLargeLabel.height / 2.0f,
194 getLargeLabelStateColor(),
195 getFaintFontColor_Dark(getSelectedKey().getNoteName()));
196 ofPopStyle();
197 }
198 else {
199 ofPushStyle();
200 ofSetColor(ofColor::white);
201 vizElm_circleInner.draw();
202 ofPopStyle();
203 }
