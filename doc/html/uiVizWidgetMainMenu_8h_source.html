262 virtual void onMenuItemSelected(uiVizWidgetMenuCollectionArgs & args) override {
263 if (args.menuTag == MENU_FILE) {
264 switch(args.activeMenuTabId) {
265 case uiVizIconCache::IconTag::WIDGET_SETTINGS:
266 switch(args.menuItem->uniqueID) {
267 case 0: // New
268 newProject(); break;
269 case 1: // save
270 saveProject(false); break;
271 case 2: // save as
272 saveProject(true); break;
273 case 3: // load
274 loadProject(); break;
275 case 4: // settings
276 settings(); break;
277 case -1000: {
278 clearExistingWidgets();
279 OF_EXIT_APP(0);
280 break;
281 }
282 }
283 break;
284 }
285 } else if (args.menuTag == MENU_HELP) {
286 switch(args.activeMenuTabId) {
287 case uiVizIconCache::IconTag::BOOK_CONTENT:
288 switch(args.menuItem->uniqueID) {
289 case 0: // tutorials
290 tutorials(); break;
291 case 1: // licence
292 licence(); break;
293 case 2: // about
294 about(1); break;
295 case 3: // about
296 about(2); break;
297 case 4: // about
298 about(3); break;
299 }
300 break;
301 }
302 } else if (args.menuTag == MENU_DEBUG) {
303 uiVizWidgetManager::drawDebugInfo(true);
304 }
